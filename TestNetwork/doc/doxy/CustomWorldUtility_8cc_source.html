237 int** CustomWorldUtility::readXML(int fileName)
238 {
239  // get the xml from the parameter, return type cXMLElement
240  cXMLElement *rootE;
241  if (fileName == xmlPressure) {
242  rootE = par("xmlPressure").xmlValue();
243  } else if (fileName == xmlTemperature) {
244  rootE = par("xmlTemperature").xmlValue();
245  } else if (fileName == xmlHumidity) {
246  rootE = par("xmlHumidity").xmlValue();
247  } else if (fileName == xmlLight) {
248  rootE = par("xmlLight").xmlValue();
249  } else {
250  throw new exception;
251  }
252 
253  // get a vector (of type cXMLElement) with all childs of the root-tag
254  cXMLElementList nListRows = rootE->getChildren();
255  int amountRows = nListRows.size();
256 
257  int** data = new int*[amountRows];
258  for (int i = 0; i < amountRows; i++){
259  //this is a row as cXMLElement
260  cXMLElement* nListRowArray = nListRows[i];
261  //this is a list of an entire row as cXMLElements
262  cXMLElementList nLeafElementsList = nListRowArray->getChildren();
263  int amountColumns = nLeafElementsList.size();
264  data[i] = new int[amountColumns];
265  for (int j = 0; j < amountColumns; j++) {
266  data[i][j] = atoi(nLeafElementsList[i]->getNodeValue());
267  }
268  }
269 
270  return data;
271 }
